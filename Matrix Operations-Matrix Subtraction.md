    matrix=[[0]*n for row in range(n)]
    for i in range(n):

        for j in range(n):
            matrix[i][j]=lines[j]
    return matrix
def print_matrix(M):
    print("Matrix:")
    for i in range(len(M)):
        for j in range(len(M[0])):
            if(i>j or i==j):
                print(M[i][j],end=" ")
